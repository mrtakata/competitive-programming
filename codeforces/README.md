Example: https://codeforces.com/contest/1118/problem/F1

## Template C++ 
```
#include <bits/stdc++.h>
using namespace std;
// define macros
#define ll long long int

int main(){

    // variable declaration
    ll n;
    ll ans;
    
    // read input
    cin >> n;

    // print answer
    cout << ans;
    return 0;
}
```
